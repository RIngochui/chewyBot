  return 1.0 / (1.0 + math.exp(-x))

Helper function — _score_leg(...) -> float:
  Inputs: recent_games: list[dict], team_stats: dict | None, american_odds: int,
          is_home: bool, leg_type: str, weight_lookup: dict[str, tuple[float, int, int]]
  
  Factor 1 — recent_form (0.25):
    Filter recent_games by team_id (passed as param), take last 5.
    Win = home_team_score > away_team_score for home team, or away > home for away.
    recent_form = wins / count if count > 0 else 0.5
  
  Factor 2 — home_away_split (0.20):
    From team_stats dict: try keys "home_record" and "away_record" ("W-L" string format).
    Parse wins/(wins+losses). Default 0.5 if key missing or format unexpected.
    Use home_record if is_home, else away_record.
  
  Factor 3 — rest_days (0.15):
    Most recent game date in recent_games (sorted descending by "date" key, ISO format).
    days = (date.today() - date.fromisoformat(last_date)).days
    rest_score = min(1.0, _sigmoid(days / 2.0))
    Default 0.5 if no recent games.
  
  Factor 4 — line_value (0.25):
    line_value_score = 1.0 / american_to_decimal(american_odds) if decimal > 0 else 0.5
    This is the implied probability — higher = more likely (shorter odds).
  
  Factor 5 — historical_hit_rate (0.15):
    weight_lookup[leg_type] = (weight, hit_count, miss_count) loaded from DB.
    hit_rate = hit_count / (hit_count + miss_count) if total > 0 else 0.5
  
  leg_score = sum of (factor_value * factor_weight for each of 5 factors)
  Return max(0.0, min(1.0, leg_score))

def build_parlay_embed(parlay: Parlay, post_date: str) -> discord.Embed:
  """Build Discord embed for daily NBA parlay post.

  PAR-10: title "chewyBot's NBA Parlay — [date]"
  Fields: one field per leg showing team, market, line, American odds.
  Summary fields: combined parlay odds (American format), confidence score 0-100.
  Footer reaction prompt. Uses EMBED_COLOR (BOT-05).
  """

  embed = discord.Embed(
      title=f"chewyBot's NBA Parlay — {post_date}",
      color=EMBED_COLOR,
  )

  # One field per leg
  for i, leg in enumerate(parlay.legs, 1):
      line_str = f" ({leg.line_value:+.1f})" if leg.line_value is not None else ""
      odds_str = f"+{leg.american_odds}" if leg.american_odds > 0 else str(leg.american_odds)
      embed.add_field(
          name=f"Leg {i}: {leg.team}",
          value=f"{leg.market_type.title()}{line_str} — {odds_str}",
          inline=False,
      )

  # Combined odds (convert decimal to American for display)
  from utils.odds_math import decimal_to_american
  combined_american = decimal_to_american(parlay.combined_odds)
  combined_str = f"+{combined_american}" if combined_american > 0 else str(combined_american)

  embed.add_field(name="Combined Parlay Odds", value=combined_str, inline=True)
  embed.add_field(name="Confidence", value=f"{parlay.confidence_score:.0f}/100", inline=True)

  embed.set_footer(text="React ✅ if this parlay hits, ❌ if it misses — helps the AI learn!")

  return embed
