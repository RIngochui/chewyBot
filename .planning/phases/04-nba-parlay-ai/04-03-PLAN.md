CLASS: ParlayCog(commands.Cog, name="Parlay")

__init__(self, bot: commands.Bot) -> None:
  self.bot = bot
  # Parse PARLAY_POST_TIME (ET timezone per CONTEXT.md Decision E)
  hour, minute = map(int, config.PARLAY_POST_TIME.split(':'))
  try:
      tz = zoneinfo.ZoneInfo("America/New_York")
  except Exception:
      tz = datetime.timezone.utc  # fallback if zoneinfo not available
  post_time = datetime.time(hour=hour, minute=minute, tzinfo=tz)
  self.daily_parlay.change_interval(time=[post_time])

async def cog_load(self) -> None:
  """Seed leg_type_weights for all 6 leg types and start daily loop. PAR-09."""
  ALL_LEG_TYPES = ["h2h_favorite", "h2h_underdog", "spread_home", "spread_away",
                   "totals_over", "totals_under"]
  async with get_db() as db:
      for leg_type in ALL_LEG_TYPES:
          await db.execute(SEED_LEG_TYPE_WEIGHTS, (leg_type,))
  logger.info("ParlayCog: leg_type_weights seeded for all 6 leg types")
  self.daily_parlay.start()

async def cog_unload(self) -> None:
  """Cancel daily task on unload."""
  self.daily_parlay.cancel()

BACKGROUND TASK — daily_parlay:
  @tasks.loop(time=datetime.time(hour=16, minute=0, tzinfo=datetime.timezone.utc))
  async def daily_parlay(self) -> None:
    """Auto-post daily parlay at PARLAY_POST_TIME. PAR-02."""
    channel = self.bot.get_channel(config.PARLAY_CHANNEL_ID)
    if not channel:
        logger.warning("ParlayCog: PARLAY_CHANNEL_ID channel not found")
        return

    parlay = await generate_parlay(min_leg_score=config.MIN_LEG_SCORE)

    if parlay is None:
        # No-games fallback: log to LOG_CHANNEL_ID, no post to PARLAY_CHANNEL_ID (Decision B)
        log_channel = self.bot.get_channel(config.LOG_CHANNEL_ID)
        if log_channel:
            await log_channel.send("[Parlay] Skipped daily post — fewer than 3 scoreable legs found")
        logger.info("ParlayCog: daily post skipped — no scoreable legs")
        return

    await self._post_and_save_parlay(channel, parlay)

  @daily_parlay.before_loop
  async def before_daily_parlay(self) -> None:
      await self.bot.wait_until_ready()

PRIVATE HELPER — _post_and_save_parlay(channel, parlay: Parlay) -> None:
  """Post parlay embed and persist to DB with legs and message_id. PAR-05."""
  post_date = datetime.now(timezone.utc).strftime("%Y-%m-%d")
  embed = build_parlay_embed(parlay, post_date)
  msg = await channel.send(embed=embed)

  # Persist parlay then legs (PAR-05)
  async with get_db() as db:
      cursor = await db.execute(INSERT_PARLAY, (
          parlay.combined_odds,
          parlay.confidence_score,
          len(parlay.legs),
          parlay.generated_at.isoformat(),
      ))
      parlay_id = cursor.lastrowid

      # Update message_id immediately
      await db.execute(UPDATE_PARLAY_MESSAGE_ID, (str(msg.id), parlay_id))

      # Insert each leg
      for leg in parlay.legs:
          await db.execute(INSERT_PARLAY_LEG, (
              parlay_id,
              leg.team,
              leg.market_type,
              leg.line_value,
              leg.american_odds,
              leg.leg_score,
              leg.leg_type,
          ))

  logger.info(f"ParlayCog: posted parlay id={parlay_id} message_id={msg.id}")

/parlay_stats — hit rate, total tracked, best/worst leg types (PAR-12):
  @app_commands.command(name="parlay_stats", description="Show NBA parlay AI performance stats")
  async def parlay_stats(self, interaction: discord.Interaction) -> None:
    await interaction.response.defer(ephemeral=True)
    async with get_db() as db:
        # Overall stats
        cursor = await db.execute(SELECT_PARLAY_STATS)
        stats_row = await cursor.fetchone()
        total_tracked = stats_row["total_tracked"] if stats_row else 0
        total_hits = stats_row["total_hits"] if stats_row else 0
        total_misses = stats_row["total_misses"] if stats_row else 0

        # Leg type weights
        cursor = await db.execute(SELECT_ALL_LEG_TYPE_WEIGHTS)
        weight_rows = await cursor.fetchall()

    hit_rate = (total_hits / total_tracked * 100) if total_tracked > 0 else 0.0

    # Sort leg types by hit rate for best/worst
    def leg_hit_rate(row):
        h, m = row["hit_count"], row["miss_count"]
        return h / (h + m) if (h + m) > 0 else 0.0

    sorted_legs = sorted(weight_rows, key=leg_hit_rate, reverse=True)
    best = sorted_legs[0]["leg_type"] if sorted_legs else "N/A"
    worst = sorted_legs[-1]["leg_type"] if sorted_legs else "N/A"

    embed = discord.Embed(title="Parlay AI Stats", color=EMBED_COLOR)
    embed.add_field(name="Total Tracked", value=str(total_tracked), inline=True)
    embed.add_field(name="Hit Rate", value=f"{hit_rate:.1f}%", inline=True)
    embed.add_field(name="Hits / Misses", value=f"{total_hits} / {total_misses}", inline=True)
    embed.add_field(name="Best Leg Type", value=best, inline=True)
    embed.add_field(name="Worst Leg Type", value=worst, inline=True)

    # Add per-leg-type breakdown
    breakdown = "\n".join(
        f"{r['leg_type']}: {leg_hit_rate(r)*100:.0f}% ({r['hit_count']}H/{r['miss_count']}M)"
        for r in sorted_legs
    )
    if breakdown:
        embed.add_field(name="Leg Type Breakdown", value=breakdown, inline=False)

    await interaction.followup.send(embed=embed, ephemeral=True)

/parlay_history [n] — last n parlays (PAR-13):
  @app_commands.command(name="parlay_history", description="Show recent parlay results")
  @app_commands.describe(n="Number of parlays to show (default 5, max 10)")
  async def parlay_history(self, interaction: discord.Interaction, n: int = 5) -> None:
    await interaction.response.defer(ephemeral=True)
    n = min(max(1, n), 10)  # clamp 1-10
    async with get_db() as db:
        cursor = await db.execute(SELECT_LATEST_PARLAYS, (n,))
        rows = await cursor.fetchall()
    if not rows:
        await interaction.followup.send("No parlays tracked yet.", ephemeral=True)
        return
    embed = discord.Embed(title=f"Parlay History (last {len(rows)})", color=EMBED_COLOR)
    for row in rows:
        outcome_emoji = {"hit": "✅", "miss": "❌", "pending": "⏳"}.get(row["outcome"], "?")
        odds_display = f"+{row['combined_odds']:.0f}" if row["combined_odds"] > 0 else f"{row['combined_odds']:.2f}x"
        embed.add_field(
            name=f"{row['generated_at'][:10]} — {outcome_emoji} {row['outcome'].upper()}",
            value=f"Legs: {row['leg_count']} | Odds: {odds_display} | Confidence: {row['confidence_score']:.0f}/100",
            inline=False,
        )
    await interaction.followup.send(embed=embed, ephemeral=True)
