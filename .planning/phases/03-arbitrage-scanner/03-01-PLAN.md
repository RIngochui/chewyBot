    american_to_decimal:
      if american_odds > 0: return (american_odds / 100) + 1.0
      else: return (100 / abs(american_odds)) + 1.0

    decimal_to_american:
      if decimal_odds >= 2.0: return round((decimal_odds - 1) * 100)
      else: return round(-100 / (decimal_odds - 1))

    implied_probability:
      return 1.0 / decimal_odds

    no_vig_probability:
      raw_probs = [1.0 / o for o in odds_list]
      total = sum(raw_probs)
      return [p / total for p in raw_probs]
