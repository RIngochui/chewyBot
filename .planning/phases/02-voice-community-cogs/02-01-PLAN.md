**MusicCog.__init__:** Store `self.bot`. Initialise per-guild state dict:
`self._guild_state: dict[int, dict] = {}` — keyed by guild_id. Helper method:
```python
def _get_state(self, guild_id: int) -> dict:
    if guild_id not in self._guild_state:
        self._guild_state[guild_id] = {
            'queue': [],          # list of song dicts
            'current_index': 0,
            'is_playing': False,
            'volume': 0.5,
            'loop': 'off',        # 'off' | 'song' | 'queue'
            'channel_id': None,   # text channel for replies
            'start_time': None,   # datetime.datetime.utcnow() when song began
        }
    return self._guild_state[guild_id]
```

**async def _fetch_song(url: str) -> dict | None:** Module-level or staticmethod. Runs
`yt_dlp.YoutubeDL(YDL_OPTS).extract_info(url, download=False)` in
`asyncio.get_event_loop().run_in_executor(None, ...)` to avoid blocking. Returns song dict
(fields: url, webpage_url, title, duration, thumbnail, requester_id=0, requester='').
Returns None on error and logs exception.

**async def _fetch_playlist(url: str) -> list[dict]:** Uses YDL_PLAYLIST_OPTS with
`noplaylist=False`. Returns list of song dicts. Each entry in `info['entries']` becomes a
song dict with `webpage_url` set to `entry.get('url', entry.get('webpage_url', ''))`.

**async def _play_next(self, guild: discord.Guild) -> None:** Core playback loop.
```python
state = self._get_state(guild.id)
vc = guild.voice_client
if vc is None or not vc.is_connected():
    state['is_playing'] = False
    return
if state['current_index'] >= len(state['queue']):
    # Queue exhausted
    state['is_playing'] = False
    await self._log_embed(guild, "queue_end")
    return
song = state['queue'][state['current_index']]
source = discord.FFmpegPCMAudio(song['url'], **FFMPEG_OPTIONS)
source = discord.PCMVolumeTransformer(source, volume=state['volume'])
state['start_time'] = datetime.datetime.utcnow()
state['is_playing'] = True

def after_play(error):
    if error:
        logger.error("Playback error in guild %d: %s", guild.id, error)
    # Advance index based on loop mode
    if state['loop'] == 'song':
        pass  # keep same index
    elif state['loop'] == 'queue' and state['current_index'] + 1 >= len(state['queue']):
        state['current_index'] = 0
    else:
        state['current_index'] += 1
    asyncio.run_coroutine_threadsafe(self._play_next(guild), self.bot.loop)

vc.play(source, after=after_play)
await self._log_embed(guild, "song_start", song)
```

**async def _log_embed(self, guild, event_type, song=None):** Sends embed to LOG_CHANNEL_ID.
- event_type "song_start": embed with title f"Now Playing: {song['title']}", URL=song['webpage_url'],
  thumbnail=song['thumbnail'], fields: Duration (seconds formatted as MM:SS), Requested By.
  Color=EMBED_COLOR.
- event_type "queue_end": embed title "Queue Finished", description "No more songs in queue.",
  color=EMBED_COLOR.
- event_type "playlist_added": song param is a dict with 'name' (playlist title), 'count' (int),
  'thumbnail' (first song thumbnail). Embed: title f"Playlist Added: {name}", fields: Tracks=count.

**Helper _format_duration(seconds: int) -> str:** Returns "MM:SS" string.

**Helper _ensure_voice(interaction) -> discord.VoiceClient | None:**
Checks `interaction.user.voice` and `interaction.user.voice.channel`. If absent, sends
ephemeral error embed "You must be in a voice channel." and returns None.
If bot is in a different voice channel than user, moves to user's channel via `await vc.move_to(channel)`.
If bot not connected, connects with `await channel.connect()`.
Returns the VoiceClient.

Create a `QueueView(discord.ui.View)` class (can be defined at module level above MusicCog):
```python
class QueueView(discord.ui.View):
    def __init__(self, queue: list[dict], current_index: int, page_size: int = 10) -> None:
        super().__init__(timeout=60)
        self.queue = queue
        self.current_index = current_index
        self.page_size = page_size
        self.current_page = 0

    def build_embed(self) -> discord.Embed:
        total_pages = max(1, -(-len(self.queue) // self.page_size))
        start = self.current_page * self.page_size
        end = start + self.page_size
        lines = []
        for i, song in enumerate(self.queue[start:end], start=start):
            marker = "▶" if i == self.current_index else f"{i + 1}."
            lines.append(f"{marker} [{song['title']}]({song['webpage_url']}) — {_format_duration(song['duration'])}")
        embed = discord.Embed(
            title="Music Queue",
            description="\n".join(lines) if lines else "Queue is empty.",
            color=EMBED_COLOR,
        )
        embed.set_footer(text=f"Page {self.current_page + 1}/{total_pages} • {len(self.queue)} tracks total")
        return embed

    @discord.ui.button(label="◀", style=discord.ButtonStyle.secondary)
    async def prev_page(self, interaction: discord.Interaction, button: discord.ui.Button) -> None:
        self.current_page = max(0, self.current_page - 1)
        await interaction.response.edit_message(embed=self.build_embed(), view=self)

    @discord.ui.button(label="▶", style=discord.ButtonStyle.secondary)
    async def next_page(self, interaction: discord.Interaction, button: discord.ui.Button) -> None:
        total_pages = max(1, -(-len(self.queue) // self.page_size))
        self.current_page = min(total_pages - 1, self.current_page + 1)
        await interaction.response.edit_message(embed=self.build_embed(), view=self)
```

In `/queue`: build `QueueView(state['queue'], state['current_index'])`. If queue empty, respond
with simple embed "Queue is empty.". Else respond with `view.build_embed()` and `view=view`.
